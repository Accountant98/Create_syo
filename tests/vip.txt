-- ===================================1. Select thong tin xe=========================================
SELECT * FROM syo_project.information_project;

-- ===================================2. Select trang thai thong tin xe =========================================
SELECT
	 value_inf.parameter_name,
	 project.project_name,
	 config.config_name,
	 value_inf.value
FROM syo_project.value_inf
INNER JOIN syo_project.project ON project.project_id = value_inf.project_id
INNER JOIN syo_project.config ON config.config_id = value_inf.config_id;

-- ===================================3. Select thong tin va trang thai cua xe =========================================

SELECT
	value_inf.parameter_name,
	information_project.keyword,
	information_project.group_infor,
	information_project.auto_infor,
	project.project_name,
	config.config_name,
	value_inf.value

FROM syo_project.value_inf
INNER JOIN syo_project.project ON project.project_id = value_inf.project_id
INNER JOIN syo_project.config ON config.config_id = value_inf.config_id
INNER JOIN syo_project.information_project ON information_project.parameter_name = value_inf.parameter_name;

-- ===================================4. Select toan bo thong tin cua devide+ device detail =========================================

SELECT
	device_details.device_details_name,
	device_details.option_detail,
	device_details.auto_detail,
	device_details.group_detail,
	device_details.device_name,
	device.device_group

FROM syo_project.device_details
INNER JOIN syo_project.device ON device.device_name = device_details.device_name;

-- ===================================5. Select toan bo thong tin device detail và trang thai tung config =========================================
SELECT
	project.project_name,
	config.config_name,
	device_details.device_details_name,
	device_details.option_detail,
	device_details.auto_detail,
	device_details.group_detail,
	device_details.device_name,
	device.device_group,
	status_config_device_detail.status

FROM syo_project.status_config_device_detail

INNER JOIN syo_project.device_details ON device_details.device_details_id = status_config_device_detail.device_details_id
INNER JOIN syo_project.config ON config.config_id = status_config_device_detail.config_id
INNER JOIN syo_project.project ON project.project_id = status_config_device_detail.project_id
INNER JOIN syo_project.device ON device_details.device_name=device.device_name ;

-- ===================================6. Select toan bo thong tin comment của tung device detail =========================================

SELECT
	project.project_name,
	device_details.device_details_name,
	comment_column.comment_name,
	project_device_comment.comment_detail

FROM syo_project.project_device_comment
INNER JOIN syo_project.device_details ON device_details.device_details_id = project_device_comment.device_details_id
INNER JOIN syo_project.project ON project.project_id = project_device_comment.project_id
INNER JOIN syo_project.comment_column ON comment_column.comment_id=project_device_comment.comment_id ;

-- ===================================7. Select toan bo thong tin lot+option code với từng config =========================================
SELECT
	project.project_name,
	config.config_name,
	lot.lot_name,
	optioncode.optioncode_value,
	status_lot_config.status

FROM syo_project.status_lot_config

INNER JOIN syo_project.project ON project.project_id = status_lot_config.project_id
INNER JOIN syo_project.config ON config.config_id = status_lot_config.config_id
INNER JOIN syo_project.optioncode ON status_lot_config.config_id = optioncode.config_id and status_lot_config.project_id = optioncode.project_id
INNER JOIN syo_project.lot ON lot.lot_id=status_lot_config.lot_id ;
